# 阶段一：版本持续维护闭环 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 让 workflow 支持从指定/最新版本继续维护，在生成新版本后完成增量索引、质量评价、修改验证和“继续优化”路由。

**Architecture:** 保留现有内容诊断和建议 Agent 的内部实现，先重构 workflow 入口和版本生命周期。新增 import/maintain/verify 三种模式、版本 lineage、确定性质量指标 + Agent 解释、Verification/Critic 路由；所有高风险动作仍经过现有人工审核。

**Tech Stack:** FastAPI、Pydantic、LangGraph、LangChain、SQLite、Qdrant、Vue 3、pytest、Node contract tests。

---

## 0. 阶段边界

本阶段必须交付：

- `base_version_id` 可以选择 v1.1/v1.2 继续维护。
- 未指定 base version 时选择该文件最新版本。
- 新版本保存 lineage，并触发增量/完整兜底索引。
- 修改前后使用同一评分版本生成质量评价。
- Verification 输出 resolved/unresolved/introduced 和质量变化。
- 用户可以选择继续优化新版本或结束。
- 任意必需节点失败不会被报告节点覆盖成 completed。
- PLAN 位置明确复用当前 `diagnosis_planning_node + DiagnosisPlanningAgent`，阶段一只执行单次静态规划；不创建空 Supervisor，也不跳过 PLAN。

本阶段不做：候选级 Send、多 Worker、split/merge 执行、模型路由、Golden set，以及 Supervisor 的滚动 plan revision/批次反馈扩圈缩圈；这些能力分别由阶段二～四交付。阶段四只替换 `diagnosis_planning_node` 内部 planning service，不改变节点名和主图输入输出契约。

```python
# backend/app/schemas/workflow.py
from typing import Literal
from pydantic import BaseModel, Field, model_validator

WorkflowMode = Literal["import", "maintain", "verify"]

class StartWorkflowRequest(BaseModel):
    mode: WorkflowMode = "import"
    file_id: int | None = None
    base_version_id: int | None = None
    result_version_id: int | None = None
    max_rounds: int = Field(default=2, ge=1, le=5)

    @model_validator(mode="after")
    def validate_scope(self) -> "StartWorkflowRequest":
        if self.mode == "import" and self.file_id is None:
            raise ValueError("import mode requires file_id")
        if self.mode == "maintain" and self.file_id is None and self.base_version_id is None:
            raise ValueError("maintain mode requires file_id or base_version_id")
        if self.mode == "verify" and (
            self.base_version_id is None or self.result_version_id is None
        ):
            raise ValueError("verify mode requires base_version_id and result_version_id")
        return self

class ContinueWorkflowDecision(BaseModel):
    decision: Literal["continue", "finish"]
    operator: str = "local_user"
```

```python
from pydantic import BaseModel
from backend.app.schemas.workflow import StartWorkflowRequest

class ResolvedWorkflowContext(BaseModel):
    mode: str
    file_id: int
    base_version_id: int | None = None
    result_version_id: int | None = None
    max_rounds: int = 2

class WorkflowContextService:
    def __init__(self, settings):
        self.settings = settings
        self.versions = VersionRepository(settings)

    def resolve(self, request: StartWorkflowRequest) -> ResolvedWorkflowContext:
        if request.mode == "import":
            if FileRepository(self.settings).get_file(request.file_id) is None:
                raise ValueError("File not found")
            return ResolvedWorkflowContext(
                mode="import", file_id=int(request.file_id), max_rounds=request.max_rounds
            )
        if request.base_version_id is not None:
            base = self.versions.get_version(request.base_version_id)
        else:
            base = self.versions.get_latest_for_file(int(request.file_id))
        if base is None:
            raise ValueError("Base version not found")
        result = None
        if request.result_version_id is not None:
            result = self.versions.get_version(request.result_version_id)
            if result is None or int(result["file_id"]) != int(base["file_id"]):
                raise ValueError("Result version does not belong to base file")
        return ResolvedWorkflowContext(
            mode=request.mode,
            file_id=int(base["file_id"]),
            base_version_id=int(base["id"]),
            result_version_id=int(result["id"]) if result else None,
            max_rounds=request.max_rounds,
        )
```

```python
def resolve_input_node(state: TaxonomyGraphState) -> StateUpdate:
    if state.workflow_mode == "import":
        return {"current_step": "resolve_input", "status": "running"}
    version_id = state.result_version_id if state.workflow_mode == "verify" else state.base_version_id
    if version_id is None:
        raise WorkflowNodeError("MISSING_VERSION_ID", "Version-driven mode requires version_id")
    version = VersionService(_runtime_settings).get_version(version_id)
    if version is None:
        raise WorkflowNodeError("VERSION_NOT_FOUND", f"Version {version_id} not found")
    return {
        "file_id": int(version["file_id"]),
        "current_version_id": int(version["id"]),
        "version_no": str(version["version_no"]),
        "current_step": "load_version_context",
        "status": "running",
    }

def route_after_resolve(state: TaxonomyGraphState) -> str:
    if state.workflow_mode == "import":
        return "parse_excel_node"
    if state.workflow_mode == "verify":
        return "index_vector_node"
    return "index_vector_node"

def route_after_required_node(state: TaxonomyGraphState) -> str:
    return "generate_failed_report_node" if state.status == "failed" else "continue"
```

- [ ] **Step 3: 实现 VerificationService**

Verification 使用 `VerificationProbeService` 在基础版本和结果版本的同一逻辑范围运行相同 detector version。范围由 `affected_node_ids` 和版本 diff 扩展：目标节点、old/new parent、这些 parent 的直接 children，以及 path 发生变化节点的全部 descendants。

```python
class VerificationFinding(BaseModel):
    issue_id: int
    detector_version: str
    issue_type: str
    logical_node_id: int | None
    normalized_evidence_key: str
    risk_level: str

    @property
    def fingerprint(self) -> tuple[str, str, int | None, str]:
        return (
            self.detector_version,
            self.issue_type,
            self.logical_node_id,
            self.normalized_evidence_key,
        )

class VerificationIssueComparator:
    def __init__(self, probe: VerificationProbeService) -> None:
        self.probe = probe

    def compare(
        self,
        base_version_id: int,
        result_version_id: int,
        affected_node_ids: list[int],
    ) -> tuple[list[int], list[int], list[int]]:
        scope = self.probe.build_symmetric_scope(
            base_version_id=base_version_id,
            result_version_id=result_version_id,
            affected_node_ids=affected_node_ids,
        )
        before = self.probe.scan(base_version_id, scope, detector_version="verify-v1")
        after = self.probe.scan(result_version_id, scope, detector_version="verify-v1")
        before_by_key = {item.fingerprint: item for item in before}
        after_by_key = {item.fingerprint: item for item in after}
        resolved_keys = before_by_key.keys() - after_by_key.keys()
        unresolved_keys = before_by_key.keys() & after_by_key.keys()
        introduced_keys = after_by_key.keys() - before_by_key.keys()
        return (
            [before_by_key[key].issue_id for key in sorted(resolved_keys, key=str)],
            [before_by_key[key].issue_id for key in sorted(unresolved_keys, key=str)],
            [after_by_key[key].issue_id for key in sorted(introduced_keys, key=str)],
        )
```

introduced 的定义是结果版本在对称复检 scope 中新出现的 fingerprint；不能直接用“结果版本全部 issue 减基础版本全部 issue”，否则会把本轮未覆盖区域误判为新增问题。

```python
class VerificationService:
    def __init__(self, settings: Settings) -> None:
        self.quality_repo = QualityRepository(settings)
        self.issue_repo = DiagnosisRepository(settings)
        self.issue_comparator = VerificationIssueComparator(
            VerificationProbeService(settings)
        )

    def verify(
        self, *, workflow_id: str, base_version_id: int,
        result_version_id: int, affected_node_ids: list[int],
        max_rounds: int, current_round: int,
    ) -> VerificationResult:
        before = self.quality_repo.get_for_workflow_version(workflow_id, base_version_id)
        after = self.quality_repo.get_for_workflow_version(workflow_id, result_version_id)
        if before is None or after is None:
            raise ValueError("Verification requires before and after evaluations")
        resolved, unresolved, introduced = self.issue_comparator.compare(
            base_version_id, result_version_id, affected_node_ids
        )
        delta = round(after.total_score - before.total_score, 1)
        introduced_records = self.issue_repo.list_by_ids(introduced)
        has_high_risk_introduced = any(
            item["risk_level"] == "high" for item in introduced_records
        )
        if has_high_risk_introduced:
            next_decision = "manual_intervention"
            status = "failed"
        elif not unresolved and delta >= 0:
            next_decision = "finish"
            status = "passed"
        elif current_round < max_rounds:
            next_decision = "ask_continue"
            status = "partially_passed"
        else:
            next_decision = "finish"
            status = "degraded"
        return VerificationResult(
            status=status, resolved_issue_ids=resolved,
            unresolved_issue_ids=unresolved, introduced_issue_ids=introduced,
            quality_delta=delta, next_decision=next_decision,
            reason=(
                f"verification={status}; quality_delta={delta}; "
                f"unresolved={len(unresolved)}; introduced={len(introduced)}"
            ),
        )
```
